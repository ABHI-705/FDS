# Display the histogram of the samples, along with the probability density function:
import matplotlib.pyplot as plt
x=np.arange(1,100.)/50.
def weib(x,n,a):
    
    return(a/n)*(x/n)**(a-1)*np.exp(-(x/n)**a)
